\newcommand{\mnote}[1]{
  \stepcounter{notecount}
  \todo[
        inline,
        caption={minipage-note},
        bordercolor=bordergreen,
        linecolor=bordergreen,
        color=lightgreen,
        fancyline
      ]
      {
        \begin{minipage}{0.9\textwidth}
           #1
        \end{minipage}
      }
      {}
}
